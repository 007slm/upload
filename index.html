Syntax

OOP style:

var uploader = new IframeUploader({
    'trigger': '#upload-icon',
    'name': 'image',
    'action': '/upload',
    'data': {'xsrf': 'hash'},
    'success': function(response) {
        alert(response);
    }
});

Chain style:

var uploader = new IframeUploader({
    'trigger': '#upload-icon',
    'name': 'image',
    'action': '/upload',
    'data': {'xsrf': 'hash'}
});
uploader.success(function(response) {
    alert(response);
});

data api support

<a id="upload" data-name="image" data-action="/upload" data-data="a=a&b=b">Upload</a>
